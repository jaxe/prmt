<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="utf-8" />

  <meta name="viewport" content="width=device-width, initial-scale=1" />

  <title>Fullscreen Slideshow (Autoplay)</title>

  <style>

    html, body { height: 100%; margin: 0; background:#000; }

    #stage { position: fixed; inset: 0; display: grid; place-items: center; background: #000; }

    #img { max-width: 100vw; max-height: 100vh; object-fit: contain; background: #000; transition: opacity .4s ease; }

  </style>

</head>

<body>

  <!-- Minimal version: no controls, no overlays, no swipes; auto-plays every 15s. -->

  <div id="stage">

    <img id="img" alt="slideshow image" />

  </div>


  <script>

    // === CONFIG ===

    const TOTAL = 476;        // total number of images

    const PAD   = 3;          // digits, e.g. 3 -> 001.jpg

    const EXT   = 'jpg';      // file extension

    const DIR   = 'images';   // folder containing your images

    const INTERVAL_MS = 15000; // 15 seconds


    // Helpers

    const pad = (n) => String(n).padStart(PAD, '0');

    const pathFor = (i) => `${DIR}/${pad(i)}.${EXT}`;


    // Sequence 1..TOTAL

    const seq = Array.from({length: TOTAL}, (_, i) => i + 1);


    // Elements

    const imgEl = document.getElementById('img');


    let idx = 0;


    function setImage(i) {

      idx = (i + seq.length) % seq.length;

      const slide = seq[idx];

      const src = pathFor(slide);

      imgEl.style.opacity = 0;

      const pre = new Image();

      pre.onload = () => {

        imgEl.src = src;

        imgEl.style.opacity = 1;

        // Preload next for smoothness

        const nxt = new Image();

        nxt.src = pathFor(seq[(idx + 1) % seq.length]);

      };

      pre.src = src;

    }


    function next() { setImage(idx + 1); }


    // Kick off

    setImage(0);

    setInterval(next, INTERVAL_MS);

  </script>

</body>

</html>
